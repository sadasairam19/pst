import java.util.*;

public class Solution {

    public static List<Integer> gradingStudents(List<Integer> grades) {
        List<Integer> roundedGrades = new ArrayList<>();

        for (int grade : grades) {
            // Rule 1: No rounding for failing grades (less than 38)
            if (grade < 38) {
                roundedGrades.add(grade);
            } else {
                // Calculate the next multiple of 5
                int nextMultipleOfFive = ((grade / 5) + 1) * 5;
                
                // Rule 2: Round up if the difference is less than 3
                if (nextMultipleOfFive - grade < 3) {
                    roundedGrades.add(nextMultipleOfFive);
                } else {
                    roundedGrades.add(grade);
                }
            }
        }
        return roundedGrades;
    }

    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        int n = scan.nextInt();
        List<Integer> grades = new ArrayList<>();
        
        for (int i = 0; i < n; i++) {
            grades.add(scan.nextInt());
        }

        List<Integer> result = gradingStudents(grades);

        for (int res : result) {
            System.out.println(res);
        }
        scan.close();
    }
}
